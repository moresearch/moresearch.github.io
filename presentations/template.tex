% Focus line
\newcommand{\focusline}[1]{%
{\color{accentorange}\Large\bfseries #1}
}

% Two column macro
\newcommand{\twocolslide}[4]{%
\begin{frame}
\frametitle{#1}
\begin{columns}
\begin{column}{0.38\textwidth}
\textbf{#2}
\vspace{0.3em}
#3
\end{column}
\begin{column}{0.58\textwidth}
#4
\end{column}
\end{columns}
\end{frame}
}

% Outline separator
\newcommand{\outlineseparator}[1]{%
\begin{frame}
\frametitle{Outline}
\centering
\vfill
{\Large
\ifnum#1=1 \textbf{Section 1}\else{\color{mdalightgrey}Section 1}\fi\\[0.6em]
\ifnum#1=2 \textbf{Section 2}\else{\color{mdalightgrey}Section 2}\fi\\[0.6em]
\ifnum#1=3 \textbf{Section 3}\else{\color{mdalightgrey}Section 3}\fi\\[0.6em]
}
\vfill
\end{frame}
}
